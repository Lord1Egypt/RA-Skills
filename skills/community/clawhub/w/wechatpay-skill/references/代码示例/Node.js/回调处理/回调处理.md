            Buffer.from(key, 'utf8'),
            nonce

        const decrypted = Buffer.concat([

            decipher.final()
        ]);

            console.error('验签失败:', error);
            return false;
        }

## 注意事项

| 注意点 | 说明 |
|--------|------|
| **必须验签** | 收到回调必须先验签，验签失败不处理业务 |
| **必须解密** | resource节点需要用APIv3密钥AES-256-GCM解密 |
| **必须幂等** | 使用out_trade_no做幂等key，防止重复处理 |
| **必须校验金额** | 回调金额必须等于本地订单金额 |
| **必须返回200** | 10秒内返回200，否则微信会重试 |
| **先返回再处理** | 建议先返回200，再异步处理业务 |
