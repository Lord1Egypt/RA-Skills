| legal_status | 法律状态查询 | 50次/分钟 |
| family | 同族专利查询 | 50次/分钟 |

**Python调用示例**：

```python
import requests
import time

class CNIPAOpenAPI:
    """CNIPA开放平台API客户端"""
    
    BASE_URL = "https://open.cnipr.com/api"
    
    def __init__(self, client_id, open_id, access_token):
        self.client_id = client_id
        self.open_id = open_id
        self.access_token = access_token
    
    def search_patent_overview(self, keyword, page=1, page_size=20):
        """
        专利概览检索 (sf1)
        
        参数:
            keyword: 检索关键词
            page: 页码
            page_size: 每页数量(最大50)
        
        返回:
            dict: 专利列表和分页信息
        """
        url = f"{self.BASE_URL}/patent/sf1"
        headers = {
            "client_id": self.client_id,
            "open_id": self.open_id,
            "access_token": self.access_token,
            "Content-Type": "application/json"
        }
        payload = {
            "keyword": keyword,
            "page": page,
            "pageSize": page_size
        }
        
        response = requests.post(url, json=payload, headers=headers)
        data = response.json()
        
        # 数据标注格式
        return {
            "data": data,
            "source": "CNIPA开放平台API",
            "query_time": time.strftime("%Y-%m-%d %H:%M:%S"),
            "validity_days": 30  # 数据有效期30天
        }
    
    def search_patent_detail(self, patent_number):
        """
        专利细览检索 (sf2)
        
        参数:
            patent_number: 专利号(如 CN101234567A)
        
        返回:
            dict: 专利详细信息
        """
        url = f"{self.BASE_URL}/patent/sf2"
        headers = {
            "client_id": self.client_id,
            "open_id": self.open_id,
            "access_token": self.access_token,
            "Content-Type": "application/json"
        }
        payload = {
            "patentNumber": patent_number
        }
        
        response = requests.post(url, json=payload, headers=headers)
        return response.json()
    
    def get_legal_status(self, patent_number):
        """
        法律状态查询
        
        参数:
            patent_number: 专利号
        
        返回:
            dict: 法律状态详情
        """
        url = f"{self.BASE_URL}/patent/legal_status"
        headers = {
            "client_id": self.client_id,
            "open_id": self.open_id,
            "access_token": self.access_token,
            "Content-Type": "application/json"
        }
        payload = {
            "patentNumber": patent_number
        }
        
        response = requests.post(url, json=payload, headers=headers)
        return response.json()
    
    def get_patent_family(self, patent_number):
        """
        同族专利查询
        
        参数:
            patent_number: 专利号
        
        返回:
            dict: 同族专利列表
        """
        url = f"{self.BASE_URL}/patent/family"
        headers = {
            "client_id": self.client_id,
            "open_id": self.open_id,
            "access_token": self.access_token,
            "Content-Type": "application/json"
        }
        payload = {
            "patentNumber": patent_number
        }
        
        response = requests.post(url, json=payload, headers=headers)
        return response.json()

# 使用示例
if __name__ == "__main__":
    # 需替换为实际凭证
```

**Python调用示例**：

```python
import requests
import time

class USPTOOpenDataAPI:
    """USPTO开放数据API客户端"""
    
    BASE_URL = "https://developer.uspto.gov"
    
    def __init__(self, api_key=None):
        self.api_key = api_key
    
    def search_patents(self, query, start=0, rows=25):
        """
        专利检索
        
        参数:
            query: 检索表达式
            start: 起始索引
            rows: 返回数量
        
        返回:
            dict: 检索结果
        """
        url = f"{self.BASE_URL}/api/v1/patents"
        params = {
            "q": query,
            "start": start,
            "rows": rows
        }
        headers = {"Accept": "application/json"}
        if self.api_key:
            headers["X-Api-Key"] = self.api_key
        
        response = requests.get(url, params=params, headers=headers)
        data = response.json()
        
        return {
            "data": data,
            "source": "USPTO Open Data Portal",
            "query_time": time.strftime("%Y-%m-%d %H:%M:%S")
        }
    
    def get_enriched_citations(self, patent_number):
        """
        获取富引文数据
        
        参数:
            patent_number: 专利号(7位数,如 1234567)
        
        返回:
            dict: 引文详情
        """
        url = f"{self.BASE_URL}/api/v1/patents/{patent_number}/citations"
        headers = {"Accept": "application/json"}
        if self.api_key:
            headers["X-Api-Key"] = self.api_key
        
        response = requests.get(url, headers=headers)
        return response.json()
    
    def get_assignments(self, patent_number):
        """
        获取转让记录
        
        参数:
            patent_number: 专利号
        
        返回:
            dict: 转让历史
        """
        url = f"{self.BASE_URL}/api/v1/assignments"
        params = {"patentNumber": patent_number}
        headers = {"Accept": "application/json"}
        if self.api_key:
            headers["X-Api-Key"] = self.api_key
        
        response = requests.get(url, params=params, headers=headers)
        return response.json()

# 使用示例
if __name__ == "__main__":
    api = USPTOOpenDataAPI()
    
    # 检索AI相关美国专利
    result = api.search_patents("artificial intelligence", rows=50)
    print(f"数据来源: {result['source']}")
    print(f"查询时间: {result['query_time']}")
    
    # 获取特定专利的引文
    citations = api.get_enriched_citations("10123456")
    print(f"引文数量: {len(citations.get('citations', []))}")
```

### 各类数据验证规范

#### A. 专利法律状态验证

| 验证项目 | 数据源 | 频率 | 有效期 |
|----------|--------|------|--------|
| 授权状态 | CNIPA pss-system | 每次操作前 | 30天 |
| 缴费状态 | CNIPA pss-system | 每次操作前 | 30天 |
| 转让记录 | USPTO Assignment API / CNIPA | 每次交易前 | 30天 |
| 许可记录 | CNIPA pss-system | 每次交易前 | 30天 |
| 质押记录 | CNIPA pss-system | 每次交易前 | 30天 |
| 无效宣告 | CNIPA pss-system | 每次交易前 | 30天 |

**验证脚本示例**：

```python
class LegalStatusValidator:
    """专利法律状态验证器"""
    
    def __init__(self, cipa_api):
        self.api = cipa_api
    
    def validate_for_transfer(self, patent_number):
        """
        转让前法律状态验证
        
        必须验证项目：
        1. 当前授权状态
        2. 年费缴费状态
        3. 是否存在质押
        4. 是否存在许可
        5. 是否有转让限制
        
        返回:
            dict: 验证报告
        """
        result = {
            "patent_number": patent_number,
            "validation_time": time.strftime("%Y-%m-%d %H:%M:%S"),
            "source": "CNIPA pss-system",
            "checks": {}
        }
        
        # 1. 获取法律状态
        legal_status = self.api.get_legal_status(patent_number)
        result["checks"]["authorization_status"] = {
            "status": legal_status.get("status"),
            "valid": legal_status.get("status") == "GRANTED",
            "expiry_date": legal_status.get("expiry_date")
        }
        
        # 2. 检查缴费状态
        result["checks"]["fee_status"] = {
            "all_paid": legal_status.get("fees_paid"),
            "next_due_date": legal_status.get("next_fee_date"),
            "valid": legal_status.get("fees_paid", False)
        }
        
        # 3. 检查质押
        result["checks"]["pledge_status"] = {
            "has_pledge": legal_status.get("has_pledge", False),
            "pledge_details": legal_status.get("pledge_details"),
            "valid": not legal_status.get("has_pledge", False)
        }
        
        # 4. 检查许可
        result["checks"]["license_status"] = {
            "has_license": legal_status.get("has_license", False),
            "license_details": legal_status.get("license_details"),
            "exclusive": legal_status.get("exclusive_license", False)
        }
        
        # 5. 综合评估
        all_valid = all([
            result["checks"]["authorization_status"]["valid"],
            result["checks"]["fee_status"]["valid"],
            not result["checks"]["pledge_status"]["has_pledge"],
            not result["checks"]["license_status"]["exclusive"]
        ])
        
        result["overall_valid"] = all_valid
        result["recommendation"] = "可转让" if all_valid else "存在障碍，需先解决"
        
        return result
```

### 自动提醒机制

```python
from datetime import datetime, timedelta

class DataValidityManager:
    """数据有效期管理器"""
    
    VALIDITY_RULES = {
        "legal_status": 30,      # 天
        "ownership": 30,
        "family_citations": 180,
        "market_data": 90,
        "technical_analysis": 180,
        "valuation": 180
    }
    
    def check_validity(self, data_record):
        """
        检查数据是否有效
        
        返回:
            dict: {is_valid, days_remaining, needs_refresh}
        """
        query_time = datetime.strptime(
            data_record["query_time"], 
            "%Y-%m-%d %H:%M:%S"
        )
        data_type = data_record["data_type"]
        validity_days = self.VALIDITY_RULES.get(data_type, 30)
        
        elapsed = (datetime.now() - query_time).days
        remaining = validity_days - elapsed
        
        return {
            "is_valid": remaining > 0,
            "days_remaining": max(0, remaining),
            "needs_refresh": remaining <= 7,  # 提前7天提醒
            "expired": remaining <= 0
        }
```

### 交叉验证逻辑

```python
def cross_validate_legal_status(patent_number, cipa_api, wipo_api):
    """
    交叉验证专利法律状态
    
    参数:
        patent_number: 专利号
        cipa_api: CNIPA API实例
        wipo_api: WIPO API实例
    
    返回:
        dict: 验证结果
    """
    # 从CNIPA获取
    cipa_status = cipa_api.get_legal_status(patent_number)
    
    # 从WIPO获取
    wipo_status = wipo_api.get_patent_info(patent_number)
    
    # 比对结果
    results_match = (
        cipa_status.get("status") == wipo_status.get("status")
    )
    
    return {
        "cipa_status": cipa_status,
        "wipo_status": wipo_status,
        "cross_validated": results_match,
        "confidence": "HIGH" if results_match else "VERIFY_MANUALLY",
        "source_1": f"CNIPA | {time.strftime('%Y-%m-%d %H:%M')}",
        "source_2": f"WIPO | {time.strftime('%Y-%m-%d %H:%M')}"
    }
```
