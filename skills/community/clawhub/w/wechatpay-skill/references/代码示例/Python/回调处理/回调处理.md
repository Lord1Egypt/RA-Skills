# 回调处理 - Python实现（Flask）

> 基于Flask实现微信支付回调通知的接收、验签、解密和处理

## 回调流程

```
微信支付POST回调 → 验签 → 解密 → 业务处理 → 返回200
```

class WechatPayNotify:
    """微信支付回调处理"""
    
    def __init__(self, mchid, serial_no, private_key, appid, api_v3_key):
        self.mchid = mchid
        self.serial_no = serial_no
        self.private_key = private_key
        self.appid = appid
        self.api_v3_key = api_v3_key
        # 微信支付平台证书下载URL
        self.platform_cert_url = "https://api.mch.weixin.qq.com/v3/certificates"

        """

        实际生产中应该：
        1. 启动时下载平台证书并缓存
        2. 证书更新时自动刷新
        
        这里简化处理，每次验签前下载
        """
        # 实际生产应该使用缓存的证书
        # 此处省略实现，请参考官方示例
        pass
    
    def verify_signature(self, body, headers):
        """
        验签
        
        Args:
            body: 请求体（原始JSON字符串）
            headers: HTTP头（包含签名信息）
        
        Returns:
            bool: 验签是否通过
        """
        timestamp = headers.get("Wechatpay-Timestamp", "")
        nonce = headers.get("Wechatpay-Nonce", "")
        signature = headers.get("Wechatpay-Signature", "")
        serial = headers.get("Wechatpay-Serial", "")
        
        # 构造待验签串
        sign_str = f"{timestamp}\n{nonce}\n{body}\n"
        
        # 获取微信支付平台公钥
        # 实际生产中应该从缓存获取，或启动时下载
        public_key = self._get_platform_public_key(serial)
        
        if not public_key:
            return False
        
        # 验签
        signature_bytes = base64.b64decode(signature)
        try:
            public_key.verify(
                signature_bytes,
                sign_str.encode(),
                padding.PKCS1v15(),
                hashes.SHA256()
            )
            return True
        except Exception:
            return False
    
    def decrypt(self, resource):
        """
        解密回调中的敏感信息
        
        Args:
            resource: 回调中的resource节点
        
        Returns:
            dict: 解密后的数据
        """
        key = self.api_v3_key.encode()
        nonce = base64.b64decode(resource["nonce"])
        ciphertext = base64.b64decode(resource["ciphertext"])
        associated_data = resource["associated_data"].encode()
        
        aesgcm = AESGCM(key)
        plaintext = aesgcm.decrypt(nonce, ciphertext, associated_data)
        return json.loads(plaintext)
    
    def handle_payment_notify(self, body, headers):
        """
        处理支付成功回调
        
        Args:
            body: 请求体
            headers: HTTP头
        
        Returns:
            dict: 解密后的支付数据
        """
        # 1. 验签
        if not self.verify_signature(body, headers):
            raise ValueError("验签失败")
        
        # 2. 解析JSON
        data = json.loads(body)
        
        # 3. 检查事件类型
        event_type = data.get("event_type")
        if event_type != "TRANSACTION.SUCCESS":
            print(f"收到非支付成功回调: {event_type}")
            return None
        
        # 4. 解密
        resource = data.get("resource")
        payment_data = self.decrypt(resource)
        
        return payment_data
    
    def handle_refund_notify(self, body, headers):
        """
        处理退款回调
        """
        # 验签
        if not self.verify_signature(body, headers):
            raise ValueError("验签失败")
        
        # 解析和解密
        data = json.loads(body)
        resource = data.get("resource")
        refund_data = self.decrypt(resource)
        
        return refund_data

# ============ Flask路由 ============

@app.route("/pay/notify", methods=["POST"])
def payment_notify():
    """
    支付成功回调通知
    """
    body = request.get_data(as_text=True)
    headers = {
        "Wechatpay-Timestamp": request.headers.get("Wechatpay-Timestamp", ""),
        "Wechatpay-Nonce": request.headers.get("Wechatpay-Nonce", ""),
        "Wechatpay-Signature": request.headers.get("Wechatpay-Signature", ""),
        "Wechatpay-Serial": request.headers.get("Wechatpay-Serial", "")
    }
    
    try:
        # 处理回调
        payment_data = notify_handler.handle_payment_notify(body, headers)
        
        if not payment_data:
            return jsonify({"code": "SUCCESS", "message": "成功"})
        
        out_trade_no = payment_data.get("out_trade_no")
        transaction_id = payment_data.get("transaction_id")
        amount = payment_data.get("amount", {}).get("payer_total", 0)
        
        # 幂等检查
        if out_trade_no in processed_orders:
            print(f"订单已处理: {out_trade_no}")
            return jsonify({"code": "SUCCESS", "message": "成功"})
        
        # 业务处理
        order_service.process_payment_success(
            transaction_id=transaction_id,
            out_trade_no=out_trade_no,
            amount=amount
        )
        
        # 标记已处理
        processed_orders.add(out_trade_no)
        
        return jsonify({"code": "SUCCESS", "message": "成功"})
    
    except Exception as e:
        print(f"处理回调异常: {e}")
        # 返回200让微信不要重试
        # 错误记录日志，人工处理
        return jsonify({"code": "SUCCESS", "message": "成功"})

## 注意事项

| 注意点 | 说明 |
|--------|------|
| **必须验签** | 收到回调必须先验签，验签失败不处理业务 |
| **必须解密** | resource节点需要用APIv3密钥AES-256-GCM解密 |
| **必须幂等** | 使用out_trade_no做幂等key，防止重复处理 |
| **必须校验金额** | 回调金额必须等于本地订单金额 |
| **必须返回200** | 10秒内返回200，否则微信会重试（最多24小时） |
| **先返回再处理** | 建议先返回200，再异步处理业务 |
