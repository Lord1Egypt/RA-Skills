# 周报生成逻辑

## 客户周报生成

```
用户问：生成本周客户周报
```

### 实现步骤

1. 计算本周时间范围（周一00:00 ~ 周日23:59）
2. 调用 feishu_bitable_list_records 获取所有记录
3. 筛选出本周创建的客户
4. 按状态分组统计
5. 生成摘要文字

### 代码逻辑

```python
from datetime import datetime, timedelta

def get_week_range():
    """获取本周一和下周一的时间戳（毫秒）"""
    today = datetime.now()
    # 本周一 00:00
    monday = today - timedelta(days=today.weekday())
    monday_start = monday.replace(hour=0, minute=0, second=0, microsecond=0)
    # 下周一 00:00（用于过滤）
    next_monday_start = monday_start + timedelta(days=7)
    
    return {
        "week_start": int(monday_start.timestamp() * 1000),
        "week_end": int(next_monday_start.timestamp() * 1000),
        "week_start_str": monday_start.strftime("%Y-%m-%d"),
        "week_end_str": (next_monday_start - timedelta(days=1)).strftime("%Y-%m-%d")
    }

def filter_this_week_clients(records, week_start_ms):
    """筛选本周新增的客户"""
    this_week = []
    
    for rec in records:
        # 飞书创建时间字段: CreatedTime (1001)
        # 如果没有这个字段，用其他方式判断
        fields = rec["fields"]
        # 多维表格的创建时间需要在fields里获取
        # 如果没有created_time字段，跳过这条判断
        
        # 简化：检查是否有"创建时间"字段
        created = fields.get("创建时间") or fields.get("fldCreatedTime")
        if created and created >= week_start_ms:
            this_week.append(rec)
    
    return this_week

def generate_weekly_report(records, week_range):
    """生成本周客户周报"""
    # 如果没有精确的创建时间字段，改用"最近跟进日期"来估算
    # 策略：本周有过跟进或新建的客户都算
    
    this_week_clients = []
    status_stats = {"潜在客户": 0, "跟进中": 0, "已成交": 0, "流失": 0}
    total_followups = 0
    
    week_start = week_range["week_start"]
    
    for rec in records:
        fields = rec["fields"]
        last_followup = fields.get("最近跟进日期")
        status = fields.get("状态", "未知")
        name = fields.get("CRM客户管理", "未知")
        
        # 统计各状态数量
        if status in status_stats:
            status_stats[status] += 1
        
        # 本周有跟进或新建的客户
        if last_followup and last_followup >= week_start:
            this_week_clients.append(name)
            total_followups += 1
    
    # 生成报告
    report_lines = [
        f"📊 **CRM客户周报**",
        f"📅 {week_range['week_start_str']} ~ {week_range['week_end_str']}\n",
        f"**本周概况**",
        f"- 本周活跃客户：{len(set(this_week_clients))} 位",
        f"- 本周跟进次数：{total_followups} 次\n",
        f"**客户状态分布**",
        f"- 潜在客户：{status_stats['潜在客户']} 位",
        f"- 跟进中：{status_stats['跟进中']} 位",
        f"- 已成交：{status_stats['已成交']} 位",
        f"- 流失：{status_stats['流失']} 位\n",
    ]
    
    return "\n".join(report_lines)
```
