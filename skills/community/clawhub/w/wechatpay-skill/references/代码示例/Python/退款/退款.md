# 退款 - Python实现

> 基于微信支付V3 API，实现退款功能

## 退款说明

- 退款支持全额或部分退款
- 每笔订单最多退款50次
- 退款金额不能超过已支付金额
- 退款后资金退回用户支付账户

## 完整代码

```python
import json
import time
import random
import string
import base64
from datetime import datetime

import requests
from cryptography.hazmat.primitives import hashes, serialization
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.backends import default_backend


class WechatPayRefund:
    """微信支付退款"""
    
    def __init__(self, mchid, serial_no, private_key, appid, api_v3_key):
        self.mchid = mchid
        self.serial_no = serial_no
        self.private_key = private_key
        self.appid = appid
        self.api_v3_key = api_v3_key
        self.base_url = "https://api.mch.weixin.qq.com"
    
    def _generate_nonce(self, length=32):
        chars = string.ascii_letters + string.digits
        return ''.join(random.choice(chars) for _ in range(length))
    
    def _sign(self, sign_str):
        private_key = serialization.load_pem_private_key(
            self.private_key.encode(),
            password=None,
            backend=default_backend()
        )
        signature = private_key.sign(
            sign_str.encode(),
            padding.PKCS1v15(),
            hashes.SHA256()
        )
        return base64.b64encode(signature).decode()
    
    def _get_authorization(self, method, url_path, body=""):
        timestamp = str(int(time.time()))
        nonce = self._generate_nonce()
        sign_str = f"{method}\n{url_path}\n{timestamp}\n{nonce}\n{body}\n"
        signature = self._sign(sign_str)
        
        return (f'WECHATPAY2-SHA256-RSA2048 '
                f'mchid="{self.mchid}",'
                f'nonce_str="{nonce}",'
                f'timestamp="{timestamp}",'
                f'serial_no="{self.serial_no}",'
                f'signature="{signature}"')
    
    def _request(self, method, url_path, body=None):
        url = self.base_url + url_path
        headers = {
            "Authorization": self._get_authorization(
                method, url_path, json.dumps(body) if body else ""
            ),
            "Content-Type": "application/json"
        }
        
        resp = requests.post(url, headers=headers, json=body)
        return resp.json()
    
    def refund(self, out_trade_no, out_refund_no, total_fee, refund_fee, 
               reason=None, notify_url=None):
        """
        申请退款
        
        Args:
            out_trade_no: 原商户订单号
            out_refund_no: 商户退款单号（退款失败重试时必须复用此单号！）
            total_fee: 原订单金额（分）
            refund_fee: 退款金额（分）
            reason: 退款原因（可选）
            notify_url: 退款结果回调地址（可选）
        
        Returns:
            dict: 退款结果
        """
        url_path = "/v3/refund/domestic/refunds"
        
        body = {
            "out_trade_no": out_trade_no,           # 原订单号
            "out_refund_no": out_refund_no,          # 退款单号（必须唯一）
            "amount": {
                "refund": refund_fee,                # 退款金额（分）
                "total": total_fee,                   # 原订单金额（分）
                "currency": "CNY"
            }
        }
        
        if reason:
            body["reason"] = reason
        
        if notify_url:
            body["notify_url"] = notify_url
        
        result = self._request("POST", url_path, body)
        
        if "refund_id" in result:
            return {"code": 0, "refund_id": result["refund_id"], "data": result}
        else:
            return {"code": 1, "error": result}
    
    def query_refund(self, out_refund_no):
        """
        查询退款状态
        
        Args:
            out_refund_no: 商户退款单号
        
        Returns:
            dict: 退款状态
        """
        url_path = f"/v3/refund/domestic/refunds/{out_refund_no}"
        headers = {
            "Authorization": self._get_authorization("GET", url_path, ""),
            "Content-Type": "application/json"
        }
        
        url = self.base_url + url_path
        resp = requests.get(url, headers=headers)
        return resp.json()
```

```python
        # 获取回调签名验签信息（从headers）

        nonce = headers.get("Wechatpay-Nonce")
        signature = headers.get("Wechatpay-Signature")
        serial = headers.get("Wechatpay-Serial")
        
        # 验签（此处省略，实际需要用微信支付公钥验签）
        # ...
        
        # 解密
        data = json.loads(body)
        resource = data.get("resource", {})
        
        key = self.api_v3_key.encode()
        nonce_bytes = base64.b64decode(resource["nonce"])
        ciphertext = base64.b64decode(resource["ciphertext"])
        associated_data = resource["associated_data"].encode()
        
        aesgcm = AESGCM(key)
        plaintext = aesgcm.decrypt(nonce_bytes, ciphertext, associated_data)
        result = json.loads(plaintext)
        
        return result

        """
        处理退款回调

            refund_status: SUCCESS/CHANGE

        refund_data = self.decrypt_refund_notify(body, headers)
        
        refund_status = refund_data.get("refund_status")  # SUCCESS/CHANGE
```

## 注意事项

| 注意点 | 说明 |
|--------|------|
| 退款单号唯一 | 退款失败重试时必须复用原out_refund_no |
| 金额限制 | 退款金额 ≤ 已支付金额 |
| 退款次数 | 每笔订单最多50次部分退款 |
| 资金退回 | 资金退回用户支付账户（微信支付/银行卡） |
| 回调处理 | 必须实现退款回调，处理退款状态变更 |

## 错误码参考

| 错误码 | 含义 |
|--------|------|
| REFUND_NOT_EXIST | 退款单不存在 |
| REFUND_AMOUNT_ERROR | 退款金额超限 |
| REFUND_COUNT_ERROR | 退款次数超限 |
| ORDER_NOT_PAID | 订单未支付 |
| REQUEST_SUCCESS | 退款申请成功（等待处理） |
