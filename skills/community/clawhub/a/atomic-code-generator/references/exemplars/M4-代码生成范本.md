```typescript
// 验证结果接口
interface ValidationResult {
  valid: boolean;
  error?: string;
}

// 用户接口
interface User {
  id: string;
  username: string;
  email: string;
}

// 验证用户名格式
function validateUsername(username: string): ValidationResult {
  if (!username || username.length < 3 || username.length > 20) {
    return { valid: false, error: '用户名长度必须在3-20字符之间' };
  }
  if (!/^[a-zA-Z0-9_]+$/.test(username)) {
    return { valid: false, error: '用户名只能包含字母、数字和下划线' };
  }
  return { valid: true };
}

// 验证密码强度
function validatePassword(password: string): ValidationResult {
  if (!password || password.length < 8) {
    return { valid: false, error: '密码长度必须大于8位' };
  }
  return { valid: true };
}

// 验证用户身份
function authenticateUser(username: string, password: string): User|null {
  // 查询数据库验证用户
  const user = queryUser(username);
  if (!user) return null;
  
  const hashedPassword = hashPassword(password);
  if (user.password !== hashedPassword) return null;
  
  return { id: user.id, username: user.username, email: user.email };
}

// 用户登录主函数
function login(username: string, password: string): User|null {
  const usernameResult = validateUsername(username);
  if (!usernameResult.valid) return null;
  
  const passwordResult = validatePassword(password);
  if (!passwordResult.valid) return null;
  
  return authenticateUser(username, password);
}
```
