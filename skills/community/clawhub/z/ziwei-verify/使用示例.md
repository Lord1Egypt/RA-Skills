## 🎯 快速开始（3个示例）

### 示例1：基本校正流程（自动模式）

**场景**：已知出生时间，希望系统自动判断是否需要校正。

```python
from datetime import datetime
from ziwei import arrange_with_packet
from ziwei_verify import calibrate

# 步骤1：生成原始命盘
original = arrange_with_packet(
    gender="M",
    birth_dt=datetime(1993, 4, 1, 14, 0),  # 1993-04-01 14:00
    location="北京"
)

# 步骤2：自动校正
result = calibrate(
    packet=original,
    birth_dt=datetime(1993, 4, 1, 14, 0),
    max_shifts=2,          # 最多偏移 ±2 时辰
    interactive=False      # 自动模式
)

# 步骤3：查看结果
if result["status"] == "CALIBRATION_DONE":
    print("✅ 校正成功！")
    print(f"原始时间：{result['metadata']['original_birth_dt']}")
    print(f"校正时间：{result['metadata']['corrected_birth_dt']}")
    print(f"偏移量：{result['metadata']['shift_hours']} 小时")
    print(f"置信度：{result['confidence']:.2%}")
elif result["status"] == "LOW_CONFIDENCE":
    print("⚠️ 置信度不足，建议使用交互模式")
    print(f"当前置信度：{result['confidence']:.2%}")
    # 提示：尝试 interactive=True
else:
    print(f"状态：{result['status']}（无需校正）")
```

**预期输出**：

```
✅ 校正成功！
原始时间：1993-04-01T14:00:00+08:00
校正时间：1993-04-01T13:00:00+08:00
偏移量：-1 小时
置信度：82.00%
```

### 示例2：交互式校正（用户参与）

**场景**：系统无法确定最佳时间，需要用户根据候选列表手动选择。

```python
from datetime import datetime
from ziwei import arrange_with_packet
from ziwei_verify import calibrate, VerificationDialogueHandler

# 1. 生成原始盘
original = arrange_with_packet(
    gender="F",
    birth_dt=datetime(1997, 9, 30, 11, 0),
    location="上海"
)

# 2. 进入交互模式
result = calibrate(
    packet=original,
    birth_dt=datetime(1997, 9, 30, 11, 0),
    max_shifts=2,
    interactive=True      # 开启交互
)

# 3. 检查是否需要交互
if result["status"] == "NEED_VERIFICATION":
    handler = VerificationDialogueHandler()

    # 3.1 展示候选表
    table = handler.present_candidates(result)
    print("请选择校正方案：")
    print(table)

    # 3.2 用户输入（模拟）
    # 实际场景中可以是 CLI 输入、Telegram 按钮、Web 前端
    user_choice = input("请输入选项编号 (0-4): ").strip()

    # 3.3 应用选择
    try:
        calibrated = handler.confirm_selection(
            session_id="user_001",
            choice_index=int(user_choice)
        )
        print(f"✅ 已应用校正：{calibrated['metadata']['corrected_birth_dt']}")
    except ValueError as e:
        print(f"❌ 选择无效：{e}")
else:
    print(f"状态：{result['status']}，无需交互")
```

**候选表示例**：

```

│ 1  │ 09:00   │ 0.75  │ 提前2个时辰   │
```

### 示例3：批量处理 + 缓存优化

**场景**：处理大量命盘（如 1000+），需要性能优化。

```python
from datetime import datetime
from ziwei import arrange_with_packet
from ziwei_verify import calibrate
from functools import lru_cache
import time

# 3.1 启用缓存（减少重复计算）
@lru_cache(maxsize=512)
def get_cached_packet(gender, birth_dt_str, location):
    """缓存排盘结果（键：gender+birth_dt+location）"""
    birth_dt = datetime.fromisoformat(birth_dt_str)
    return arrange_with_packet(gender, birth_dt, location)

# 3.2 批量数据
batch = [
    {"gender": "M", "birth_dt": "1993-04-01T14:00:00", "location": "北京"},
    {"gender": "F", "birth_dt": "1997-09-30T11:00:00", "location": "上海"},
    # ... 共 1000 条
]

start = time.time()
results = []

for i, case in enumerate(batch):
    # 使用缓存排盘
    packet = get_cached_packet(
        case["gender"],
        case["birth_dt"],
        case["location"]
    )

    # 校正（自动模式）
    calibrated = calibrate(
        packet=packet,
        birth_dt=datetime.fromisoformat(case["birth_dt"]),
        max_shifts=2,
        interactive=False
    )

    results.append({
        "id": i,
        "original_confidence": packet["confidence"],
        "final_confidence": calibrated["confidence"],
        "calibrated": calibrated["metadata"].get("calibration_applied", False)
    })

    if (i + 1) % 100 == 0:
        print(f"已处理 {i + 1}/{len(batch)}")

elapsed = time.time() - start
print(f"批量处理完成：{len(batch)} 条，耗时 {elapsed:.2f} 秒")
print(f"平均每条：{elapsed/len(batch)*1000:.1f} ms")

# 3.3 统计校正率
calibrated_count = sum(1 for r in results if r["calibrated"])
print(f"校正率：{calibrated_count / len(batch):.2%}")
```

**性能预期**（1000条，缓存命中率 60%）：

```
批量处理完成：1000 条，耗时 45.32 秒
平均每条：45.3 ms
校正率：23.50%
缓存命中率：60.2%
```

## 🎭 常见使用模式

### 模式1：始终不校正（快速解读）

```python
result = calibrate(packet, interactive=False)
# 无论置信度多低，都保持原盘解读
report = interpret_with_packet(result)
```

**适用**：用户坚持原时间，或时间已知非常准确。

---

### 模式2：始终校正（追求最优）

```python
result = calibrate(packet, interactive=False)
if result["status"] == "LOW_CONFIDENCE":
    # 强制交互模式，不放过任何校正机会
    result = calibrate(packet, interactive=True)
    # 然后由用户选择
```

**适用**：命理师手动分析，愿意花时间选最优。

---

### 模式3：条件校正（阈值控制）

```python
AUTO_THRESHOLD = 0.75   # 自动校正阈值（更高）
INTERACTIVE_THRESHOLD = 0.6  # 交互阈值

result = calibrate(packet)

if result["confidence"] >= AUTO_THRESHOLD:
    print("✅ 置信度足够，直接解读")
    report = interpret_with_packet(result)
elif result["confidence"] >= INTERACTIVE_THRESHOLD:
    print("🔍 建议交互校正")
    # 调用交互模式
else:
    print("⚠️ 置信度过低，请检查出生时间")
```

## 📊 结果解析

### 解读 `status` 字段

```python
status = result["status"]

if status == "SUCCESS":
    # 高置信度，无需校正
    print("命盘置信度高，可直接解读")

elif status == "NEED_VERIFICATION":
    # 需要用户核对验证点
    points = result["verification_points"]
    print(f"发现 {len(points)} 个需关注的验证点：")
    for p in points:
        if p["impact"] == "high":
            print(f"  🔴 {p['field']}: {p['description']}")

elif status == "LOW_CONFIDENCE":
    # 置信度过低
    print(f"置信度仅 {result['confidence']:.1%}，建议校正")
    # 查看候选（即使 interactive=False）
    candidates = result.get("data", {}).get("all_candidates", [])
    if candidates:
        best = max(candidates, key=lambda c: c["score"])
        print(f"最优候选：{best['birth_dt']}（分数 {best['score']:.1%}）")

elif status == "CALIBRATION_DONE":
    # 已校正
    meta = result["metadata"]
    print(f"✅ 已校正：{meta['original_birth_dt']} → {meta['corrected_birth_dt']}")
    print(f"偏移：{meta['shift_hours']} 小时，置信度：{result['confidence']:.1%}")
```

### 提取校正信息

```python
def extract_calibration_info(result):
    """提取校正详情"""
    meta = result.get("metadata", {})

    if not meta.get("calibration_applied"):
        return None

    return {
        "original": meta.get("original_birth_dt"),
        "corrected": meta.get("corrected_birth_dt"),
        "shift_hours": meta.get("shift_hours"),
        "confidence": result.get("confidence"),
        "history": result.get("calibration_history", [])
    }

info = extract_calibration_info(result)
if info:
    print(f"校正历史：")
    for record in info["history"]:
        print(f"  {record['timestamp']}: {record['shift_hours']}h → {record['score']:.1%}")
```

## 🧩 与其他模块集成

### 集成 `ziwei_interpret`

```python
# 完整流水线
from ziwei import arrange_with_packet
from ziwei_verify import calibrate
from ziwei_interpret import interpret_with_packet

def full_pipeline(gender, birth_dt, location, auto_calibrate=True):
    # L1: 排盘
    packet = arrange_with_packet(gender, birth_dt, location)

    # L2: 校正
    if auto_calibrate:
        calibrated = calibrate(packet, birth_dt, interactive=False)
    else:
        calibrated = packet  # 跳过校正

    # L3: 解读
    report = interpret_with_packet(calibrated)
    return report

report = full_pipeline("M", datetime(1993, 4, 1, 14, 0), "北京")
print(report)
```

### 集成到 Web API（Flask 示例）

```python
from flask import Flask, request, jsonify
from datetime import datetime
from ziwei import arrange_with_packet
from ziwei_verify import calibrate

app = Flask(__name__)

@app.route("/api/ziwei/analyze", methods=["POST"])
def analyze():
    data = request.json
    gender = data["gender"]
    birth_dt = datetime.fromisoformat(data["birth_dt"])
    location = data["location"]
    auto_calibrate = data.get("auto_calibrate", True)

    # 排盘
    packet = arrange_with_packet(gender, birth_dt, location)

    # 校正（可选）
    if auto_calibrate:
        result = calibrate(packet, birth_dt, interactive=False)
    else:
        result = packet

    return jsonify({
        "trace_id": result["trace_id"],
        "confidence": result["confidence"],
        "status": result["status"],
        "calibration": result.get("metadata", {}),
        "verification_points": result.get("verification_points", [])
    })

if __name__ == "__main__":
    app.run(debug=True)
```

### 处理低置信度

```python
def smart_calibrate(packet, birth_dt):
    """智能校正：自动模式下若分数低则降级为交互"""
    result = calibrate(packet, birth_dt, interactive=False)

    if result["status"] == "LOW_CONFIDENCE":
        print("自动校正效果不佳，切换到交互模式...")
        # 返回交互状态，由上层处理
        return calibrate(packet, birth_dt, interactive=True)

    return result
```

## 📈 性能调优技巧

### 1. 缓存命盘结果

```python
from functools import lru_cache
from ziwei import arrange_with_packet

@lru_cache(maxsize=1024)
def cached_arrange(gender, birth_dt_iso, location):
    dt = datetime.fromisoformat(birth_dt_iso)
    return arrange_with_packet(gender, dt, location)

# 相同输入直接从内存返回，无需重算
```

---

### 2. 减少校正频率

```python
# 仅当置信度低于阈值时才校正
def conditional_calibrate(packet, birth_dt, threshold=0.5):
    if packet["confidence"] >= threshold:
        return packet  # 跳过校正

    return calibrate(packet, birth_dt)
```

---

### 3. 并行处理批量任务

```python
import asyncio
from ziwei_verify import calibrate

async def batch_calibrate_async(packets, birth_dts):
    tasks = []
    for pkt, dt in zip(packets, birth_dts):
        task = asyncio.to_thread(calibrate, pkt, dt, False)
        tasks.append(task)

    results = await asyncio.gather(*tasks)
    return results
```

## 📚 完整案例：1971年男命

**背景**：1971年3月29日 07:00 生人，原命盘置信度低，需校正。

```python
from datetime import datetime
from ziwei import arrange_with_packet
from ziwei_verify import calibrate

# 原始命盘
original = arrange_with_packet(
    gender="M",
    birth_dt=datetime(1971, 3, 29, 7, 0),
    location="上海"
)

print(f"原始置信度：{original['confidence']:.1%}")

# 校正
result = calibrate(
    packet=original,
    birth_dt=datetime(1971, 3, 29, 7, 0),
    max_shifts=2,
    interactive=False
)

print(f"最终状态：{result['status']}")
print(f"校正后置信度：{result['confidence']:.1%}")

if result["metadata"].get("calibration_applied"):
    orig = result["metadata"]["original_birth_dt"]
    corr = result["metadata"]["corrected_birth_dt"]
    shift = result["metadata"]["shift_hours"]
    print(f"校正：{orig} → {corr}（偏移 {shift}h）")
```

**输出**：

```
原始置信度：38.2%
最终状态：CALIBRATION_DONE
校正后置信度：82.1%
校正：1971-03-29T07:00:00+08:00 → 1971-03-29T05:00:00+08:00（偏移 -2h）
```

**解读**：原时间07:00校正为05:00，发现"日月并明"格局，置信度从 38% 提升至 82%。
