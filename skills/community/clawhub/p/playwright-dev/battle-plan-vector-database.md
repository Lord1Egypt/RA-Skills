const lancedb = require('@lancedb/lancedb');
const aliyun = require('aliyun-embedding-api');

async function vectorize(text) {
  // 使用阿里云 Embedding API
  const embedding = await aliyun.embed(text);
  
  // 存储到 LanceDB
  const db = await lancedb.connect('./data/vectors');
  const table = await db.createTable('docs', [
    { id: 1, text: text, vector: embedding }
  ]);
  
  return embedding;
}

### 5.2 缓解措施

**风险 1：社区技能不兼容 OpenClaw**
```
缓解措施：
1. 提前读取 SKILL.md，确认技术栈
2. 小规模测试核心功能（向量化 + 检索）
3. 准备备用方案：自建核心模块
4. 联系作者询问 OpenClaw 兼容性
```

**风险 2：阿里云 API 调用失败**
```
缓解措施：
1. 提前测试 API Key 有效性
2. 准备本地 Embedding 模型（如 sentence-transformers）
3. 实现 API 降级策略（云端→本地）
4. 监控 API 调用配额
```

**风险 3：LanceDB 安装失败**
```
缓解措施：
1. 使用预编译包（npm install @lancedb/lancedb）
2. 准备 Docker 方案（lancedb/lancedb 镜像）
3. 备选向量库：Chroma（更轻量）
4. 查阅 LanceDB 官方文档
```

**风险 4：适配时间超出预期**
```
缓解措施：
1. 分阶段验证（每阶段≤2 小时）
2. 设定时间上限，超时则调整方案
3. 优先实现核心功能（向量化 + 检索）
4. 次要功能（评估、优化）后续迭代
```
