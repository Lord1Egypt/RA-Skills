# 退款 - Node.js实现

> 基于微信支付V3 API，实现退款功能

## 完整代码

```javascript
const crypto = require('crypto');
const axios = require('axios');
const fs = require('fs');

class WechatPayRefund {
    constructor(config) {
        this.mchid = config.mchid;
        this.serialNo = config.serialNo;
        this.privateKey = config.privateKey;
        this.appid = config.appid;
        this.apiV3Key = config.apiV3Key;
        this.baseURL = 'https://api.mch.weixin.qq.com';
    }

    generateNonce(length = 32) {
        const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789';
        let result = '';
        for (let i = 0; i < length; i++) {
            result += chars.charAt(Math.floor(Math.random() * chars.length));
        }
        return result;
    }

    sign(signStr) {
        const sign = crypto.sign('RSA-SHA256', Buffer.from(signStr), this.privateKey);
        return sign.toString('base64');
    }

    getAuthorization(method, urlPath, body = '') {
        const timestamp = Math.floor(Date.now() / 1000).toString();
        const nonce = this.generateNonce();
        const signStr = `${method}\n${urlPath}\n${timestamp}\n${nonce}\n${body}\n`;
        const signature = this.sign(signStr);

        return `WECHATPAY2-SHA256-RSA2048 mchid="${this.mchid}",nonce_str="${nonce}",timestamp="${timestamp}",serial_no="${this.serialNo}",signature="${signature}"`;
    }

    async request(method, urlPath, body = null) {
        const url = this.baseURL + urlPath;
        const headers = {
            'Authorization': this.getAuthorization(method, urlPath, body ? JSON.stringify(body) : ''),
            'Content-Type': 'application/json'
        };

        const response = await axios.post(url, body, { headers });
        return response.data;
    }

    /**
     * 申请退款
     * @param {string} outTradeNo - 原商户订单号
     * @param {string} outRefundNo - 商户退款单号（退款失败重试时必须复用！）
     * @param {number} totalFee - 原订单金额（分）
     * @param {number} refundFee - 退款金额（分）
     * @param {string} reason - 退款原因
     * @param {string} notifyUrl - 退款结果回调地址
     */
    async refund(outTradeNo, outRefundNo, totalFee, refundFee, reason = null, notifyUrl = null) {
        const urlPath = '/v3/refund/domestic/refunds';

        const body = {
            out_trade_no: outTradeNo,
            out_refund_no: outRefundNo,      // 必须唯一！
            amount: {
                refund: refundFee,           // 退款金额
                total: totalFee,             // 原订单金额
                currency: 'CNY'
            }
        };

        if (reason) {
            body.reason = reason;
        }

        if (notifyUrl) {
            body.notify_url = notifyUrl;
        }

        try {
            const result = await this.request('POST', urlPath, body);
            return { code: 0, refundId: result.refund_id, data: result };
        } catch (error) {
            return { code: 1, error: error.response?.data || error.message };
        }
    }

    /**
     * 查询退款状态
     */
    async queryRefund(outRefundNo) {
        const urlPath = `/v3/refund/domestic/refunds/${outRefundNo}`;
        const headers = {
            'Authorization': this.getAuthorization('GET', urlPath, ''),
            'Content-Type': 'application/json'
        };

        const url = this.baseURL + urlPath;
        const response = await axios.get(url, { headers });
        return response.data;
    }
}


// ============ 使用示例 ============

async function main() {
    const config = {
        mchid: '1234567890',
        serialNo: 'XXXXXXXXXXXXXXXXXXXXXXXX',
        privateKey: fs.readFileSync('apiclient_key.pem', 'utf8'),
        appid: 'wxa5f5c1d6e8f9a2b3',
        apiV3Key: '0123456789abcdef0123456789abcdef'
    };

    const refund = new WechatPayRefund(config);

    // ============ 申请退款 ============
    // 注意：退款失败重试时，outRefundNo必须复用原单号！
    const result = await refund.refund(
        'ORDER123456789',           // 原订单号
        `REFUND${Date.now()}`,      // 退款单号（首次生成）
        100,                         // 原订单金额（分）
        50,                          // 退款金额（分）
        '用户申请退款',
        'https://yourdomain.com/pay/refund-notify'
    );

    if (result.code === 0) {
        console.log('退款申请成功，refund_id:', result.refundId);

        // ============ 查询退款状态 ============
        const status = await refund.queryRefund(result.refundId);
        console.log('退款状态:', status);
        /*
        {
            refund_id: "50320001752022041508580116507",
            out_refund_no: "REFUND123456789",
            refund_status: "PROCESSING",  // PROCESSING/SUCCESS/CLOSED
            amount: {
                total: 100,
                refund: 50,
                currency: "CNY"
            }
        }
        */
    } else {
        console.error('退款失败:', result.error);
    }
}

main();
```

## Express路由示例

```javascript
const express = require('express');
const router = express.Router();
const WechatPayRefund = require('./WechatPayRefund');

const refund = new WechatPayRefund({
    mchid: '1234567890',
    serialNo: 'XXXXXXXXXXXXXXXXXXXXXXXX',
    privateKey: fs.readFileSync('apiclient_key.pem', 'utf8'),
    appid: 'wxa5f5c1d6e8f9a2b3',
    apiV3Key: '0123456789abcdef0123456789abcdef'
});

/**
 * 申请退款
 * POST /api/refund/apply
 */
router.post('/apply', async (req, res) => {
    const { outTradeNo, totalFee, refundFee, reason } = req.body;

    // 生成退款单号（注意：重试时必须复用！）
    const outRefundNo = `REFUND_${outTradeNo}_${Date.now()}`;

    try {
        const result = await refund.refund(
            outTradeNo,
            outRefundNo,
            totalFee,
            refundFee,
            reason,
            'https://yourdomain.com/pay/refund-notify'
        );

        if (result.code === 0) {
            res.json({ success: true, refundId: result.refundId });
        } else {
            res.status(400).json({ success: false, error: result.error });
        }
    } catch (error) {
        console.error('退款申请失败:', error);
        res.status(500).json({ success: false, error: '服务器错误' });
    }
});

/**
 * 查询退款状态
 * GET /api/refund/query/:outRefundNo
 */
router.get('/query/:outRefundNo', async (req, res) => {
    try {
        const result = await refund.queryRefund(req.params.outRefundNo);
        res.json({ success: true, data: result });
    } catch (error) {
        console.error('查询退款失败:', error);
        res.status(500).json({ success: false, error: '服务器错误' });
    }
});

module.exports = router;
```

## 注意事项

| 注意点 | 说明 |
|--------|------|
| 退款单号唯一 | 退款失败重试时必须复用原outRefundNo |
| 金额限制 | 退款金额 ≤ 已支付金额 |
| 退款次数 | 每笔订单最多50次部分退款 |
| 资金退回 | 资金退回用户支付账户 |
| 回调处理 | 必须实现退款回调 |
