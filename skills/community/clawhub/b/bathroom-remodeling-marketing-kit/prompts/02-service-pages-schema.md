### JSON-LD SCHEMA

```json
  "url": "[WEBSITE URL]",
  "telephone": "[PHONE]",
  "address": {
    "@type": "PostalAddress",
    "addressLocality": "[CITY]",
    "addressRegion": "[STATE]",
    "postalCode": "[ZIP]"
  },
  "areaServed": ["[CITY 1]", "[CITY 2]", "[CITY 3]"],
  "hasCredential": [
    "[STATE] B-2 General Contractor #[LICENSE #]",

    "@type": "AggregateRating",

```
